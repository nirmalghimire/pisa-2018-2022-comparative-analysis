# Calculate mean scores by country and year (using sampling weights)
country_trends <- summary_data %>%
  filter(cnt != "VNM") %>%
  group_by(cnt, year) %>%
  summarise(
    mean_reading = weighted.mean(full_sample_reading_score, w_fstuwt, na.rm = TRUE),
    mean_math = weighted.mean(full_sample_math_score, w_fstuwt, na.rm = TRUE),
    mean_science = weighted.mean(full_sample_science_score, w_fstuwt, na.rm = TRUE),
    n = n()
  ) %>%
  ungroup()

# Calculate changes in scores
country_changes <- country_trends %>%
  group_by(cnt) %>%
  arrange(year) %>%
  summarise(
    reading_change = last(mean_reading) - first(mean_reading),
    math_change = last(mean_math) - first(mean_math),
    science_change = last(mean_science) - first(mean_science)
  ) %>%
  arrange(desc(math_change)) # Sort by math change in decreasing order

# Create visualization of changes
score_changes_long <- country_changes %>%
  pivot_longer(
    cols = ends_with("change"),
    names_to = "subject",
    values_to = "change"
  ) %>%
  mutate(subject = case_when(
    subject == "reading_change" ~ "Reading",
    subject == "math_change" ~ "Mathematics",
    subject == "science_change" ~ "Science"
  ))

# Plot changes for all countries in decreasing order
changes_plot <- ggplot(
  score_changes_long %>%
    group_by(cnt) %>%
    mutate(avg_change = mean(change)) %>%
    ungroup() %>%
    mutate(cnt = reorder(cnt, avg_change)), # Reorder based on average change
  aes(x = cnt, y = change, fill = subject)
) +
  geom_bar(stat = "identity", position = "dodge") +
  coord_flip() +
  theme_minimal() +
  theme(
    panel.background = element_rect(fill = "white", color = NA),
    plot.background = element_rect(fill = "white", color = NA),
    panel.grid.major = element_line(color = "grey90"),
    panel.grid.minor = element_line(color = "grey90")
  ) +
  scale_fill_manual(values = c("#003366", "#006666", "#f3e389")) +
  labs(
    title = "Changes in Test Scores by Country",
    subtitle = "Countries Ordered by Average Score Change (Decreasing)",
    x = "Country",
    y = "Change in Score",
    fill = "Subject"
  )

# Display the plot
print(changes_plot)
